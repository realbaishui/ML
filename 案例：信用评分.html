def M_dumm(dfname, collist = [], namesuffix = '', drop = True):
    ''' 作者：张文彤；最后修改：2019.3.25
    将分类变量转换为哑变量组，对两分类变量则只生成一个哑变量
    dfname
    collist
    namesuffix：哑变量名需要增加的前缀
    drop：是否删除原变量
    '''
    dfnew = copy.copy(dfname)
    if len(collist) == 0: # 准备变量名列表
        collist = dfname.columns
        
    binlist = []
    for colname in collist: # 筛检出两分类变量
        if len(set(dfname[colname])) == 2: 
            binlist.append(colname)
            collist.remove(colname)

    prefixlist = []
    for colname in collist: # 准备哑变量前缀
        prefixlist.append(colname + namesuffix)

    binprefixlist = []
    for colname in binlist: # 准备两分类变量的哑变量前缀
        binprefixlist.append(colname + namesuffix)

    if len(collist) > 0:
        dfdumm = pd.get_dummies(dfname[collist].astype('str'), prefix = prefixlist)
        dfnew = pd.merge(dfnew, dfdumm, left_index = True, right_index = True)

    if len(binlist) > 0:
        dfdumm = pd.get_dummies(dfname[binlist].astype('str'), 
                                drop_first = True, prefix = binprefixlist)
        dfnew = pd.merge(dfnew, dfdumm, left_index = True, right_index = True)

    if drop:
        dfnew.drop(columns = collist, inplace = True)
        dfnew.drop(columns = binlist, inplace = True)

    return dfnew

def M_VarBin(dfname, collist = [], binmethod = 'cut', 
             binnum = 10, keepvar = True, iscat = True, todumm = False):
    ''' 作者：张文彤；最后修改：2019.5.29
    将数据框中符合要求的连续变量进行分段
    dfname：数据框名称
    collist：变量列表，不指定则为全部符合要求的变量
    binmethod：cut 按数值等距，qcut 按频数等量，也可自定义分段方式
    binnum：分段数
    keepvar：是否保留原始变量
    iscat：转换为cat类型，否则为流水编码数值
    todumm：转换为哑变量组，并删除生成的分段变量
    '''
    
    dfnew = copy.copy(dfname)
    
    if len(collist) == 0:
        collist = dfnew.columns
        
    catlabel = None
    if not iscat: # 要求转换后变量为数值而非分类
            catlabel = range(binnum)
            
    for var in collist:
        if binmethod == 'cut':
            dfnew[var + '_bin'] = pd.cut(dfnew[var], bins = binnum,
                                    labels = catlabel, duplicates = 'drop')
            if todumm:
                dfnew = M_dumm(dfnew, [var + '_bin']) # 将变量转换为哑变量组
            elif not iscat: # 该语句用于解决pandas的一个bug
                dfnew[var + '_bin'] = dfnew[var + '_bin'].astype('int')
        elif binmethod == 'qcut':
            dfnew[var + '_qbin'] = pd.qcut(dfnew[var], q = binnum, 
                                    labels = catlabel, duplicates = 'drop')
            if todumm:
                dfnew = M_dumm(dfnew, [var + '_qbin'])
            elif not iscat:
                dfnew[var + '_qbin'] = dfnew[var + '_qbin'].astype('int')

    if not keepvar: # 删除原始变量
        dfnew.drop(columns = collist, inplace = True)

    return dfnew

def M_VarBin(dfname, collist = [], binmethod = 'cut',
             binnum = 10, keepvar = True, iscat = True, todumm = False):
    """作者：孙成超；最后修改：2019.8.23
    由学员提供的，代码更简化的连续变量分段函数，代码更简洁，但涉及了一些复杂用法   
    param dfname: pd.DataFrame 
    param collist : 变量列表，不指定则为全部符合要求的变量  
    param binmethod: cut等距 / qcut等频   
    param keepvar : bool 是否保留原始变量  
    param iscat: bool 转换为cat类型， 否则为流水编码数值  
    param todumm: bool 转换为哑变量，并删除生成的分段变量  
    """
    dfnew = copy.deepcopy(dfname)

    collist = dfnew.columns  if collist == [] else collist
    # 是否转换成数值
    catlabel = list(range(binnum)) if not iscat else None

    for var in collist:
        cut_bin = '{}_bin'.format(var)
        try:
            bin_method = eval('pd.{}'.format(binmethod))
            dfnew[cut_bin] = bin_method(dfnew[var], bins = binnum,
                                        labels = catlabel, duplicates = 'drop')
            if todumm:
                dfnew = M_dumm(dfnew, [cut_bin])
            elif not iscat:
                dfnew[cut_bin] = dfnew[cut_bin].astype('int')
        except: print("Enter correct bin method: cut / qcut")
    if not keepvar:
        dfnew.drop(columns = collist, inplace = True)
    return dfnew

def M_ChiMerge(dfname, colname, target, method = 'MaxInt', maxint = 5, minchi2 = 3.84, 
               vartype = 'N', exclude = [], outtype = 1, output = True):
    '''作者：张文彤；最后修改：2019.7.12
    使用卡方分割法对变量分箱
    dfname
    colname：需要考虑分箱的自变量名称，允许存在缺失值（不做合并）
    target：两分类因变量名称，必须已清理完毕缺失值，且为0、1两分类（程序略加修改即可分析多类）
    method：使用的分箱方法，'MaxInt' 最大分箱数；'MinChi2' 最小卡方值
    maxint：最大分箱数时希望拆分为的箱体数
    minchi2：最小卡方分割时使用的卡方界值
    vartype：变量类型，N 数值 C 类别
    exclude：不考虑进行合并的特殊取值列表，不需要包括nan
    outtype：返回结果，1 转换对应表 2 转换后原始数据列
    output 是否要求打印输出分箱结果
    返回：数值替换对应df
    '''
    
    dfana = copy.copy(dfname[[colname, target]])
    
    for item in exclude:
        if dfana[dfana[colname] == item][target].mean() in [0, 1]:
            print('警告：', colname, '的数值', item, '组坏样本率为0/1，必须进行合并。')
            exclude.remove(item)
            
    if len(exclude) > 0: # 指定了不进行合并的特殊取值
        print("警告：", colname, "中下列取值将不进行合并", exclude)
        dfana = dfana[dfana[colname].apply(lambda x : x not in exclude)] # 只保留其余行
    
    if sum(dfana[colname].isna()) > 0: # 变量中存在缺失值
        if dfana[dfana[colname].isna()][target].mean() in [0, 1]:
            print("警告：", colname, "缺失值组坏样本率为0/1，请手工进行合并处理！")
            if outtype == 1:
                return []  
            elif outtype == 2:
                return dfname[colname] # 直接返回原始序列
        else:
            print("警告：", colname, "中存在缺失值，将只对非缺失数据进行合并处理！")
            exclude.append(np.nan)
            dfana = dfana[dfana[colname].notna()] # 只保留变量无缺失值的行

    N_levels = len(set(dfana[colname]))
    
    if N_levels == 1 or (method == 'MaxInt' and N_levels <= maxint): 
        print(colname, "取值类别太少，无需分箱！")
        if outtype == 1:
            return []  
        elif outtype == 2:
            return dfname[colname] # 直接返回原始序列

    dfana['anacol'] = dfana[colname]
    
    needtrans = False # 是否需要结果反向转换
    if N_levels > 100 and vartype == 'N': # 数值取值类别数大于100时进行合并，以减少计算量 
        colLevels = sorted(list(set(dfana[colname])))
        ind_x = [int(i / 100.0 * N_levels) for i in range(1, 100)] # 定位拆分点的序号
        split_x = [colLevels[i] for i in ind_x] # 取出对应的拆分点数值
        split_x.insert(0, -np.Inf)
        split_x.append(max(dfana[colname])) # 序列加入头尾两端的端点
        split_x = sorted(list(set(split_x)))
        dfana['anacol'] = pd.cut(dfana[colname], bins = split_x, 
                                labels = split_x[1:]).astype('int') # 替换数值
        N_levels = len(set(dfana['anacol']))
        needtrans = True
        
    # 生成计算所需的交叉表
    dftbl = pd.crosstab(dfana['anacol'], dfana[target])
    dftbl.reset_index(inplace = True)
    dftbl['badrate'] = dftbl[1] / (dftbl[0] + dftbl[1])
        
    if vartype == 'C': # 分类变量应当只能和坏样本率最接近的合并，数据重排
        dftbl.sort_values('badrate', inplace = True)
        dftbl.reset_index(drop = True, inplace = True)
        print(dftbl) # 分类变量合并要慎重，因此输出数据备查
        
    dfres = pd.DataFrame({colname : dftbl['anacol'], 'trans' : dftbl['anacol']}) # 初始化结果表
    
    # 提前合并0%和100%坏样本率的组
    normalbinindex = dftbl.query("badrate > 0 and badrate < 1").index
    if len(normalbinindex) < N_levels: # 存在异常组，需要处理
        initpos = normalbinindex[0] # 定位第一个正常组的位置
        if initpos > 0 : # 处理在正常组之前的异常组
            dfres.loc[0 : initpos, 'trans'] = dfres.loc[initpos, 'trans']
        for i in range(initpos, N_levels): # 按顺序处理其他异常组
            if dftbl.loc[i, 'badrate'] == 0 or dftbl.loc[i, 'badrate'] == 1:
                dfres.loc[i, 'trans'] = dfres.loc[i - 1, 'trans']
           
        dftbl['anacol'] =  dfres.trans
        dftbl = dftbl.groupby('anacol', as_index = False, 
                              observed = True).agg('sum') # 合并对应的类别
        N_levels = len(set(dftbl['anacol'])) # 更新合并后的类别长度
        
        if vartype == 'C': # 分类变量应当只能和坏样本率最接近的合并，数据重排
            dftbl['badrate'] = dftbl[1] / (dftbl[0] + dftbl[1])
            dftbl.sort_values('badrate', inplace = True)
            dftbl.reset_index(drop = True, inplace = True)

    for i in range(N_levels - 1): # 计算相邻单元格卡方值，这里计算的是Yates校正后的卡方值
        dftbl.loc[i, 'chi2'] = ss.chi2_contingency(dftbl.loc[i : i + 1, [0, 1]])[0] 
        
    while N_levels > 2: # 这里改为N_levels > 2是确保只有2类时自动停止
        # 需要时上面的停止条件可以直接改为 > MaxInt，使得MinChi2时双条件同时生效
        minindex = dftbl.index[dftbl.chi2 == min(dftbl.chi2)][0] # 结果表中最小卡方值索引位置
        mincat = dftbl.loc[minindex, 'anacol'] # 最小卡方值对应的类别

        if method == 'MaxInt':
            if dftbl.shape[0] <= maxint: 
                break
        else:
            if dftbl.loc[minindex, 'chi2'] > minchi2:
                break
        
        # 更新计算表和结果表中相应信息
        newcat = dftbl.loc[minindex + 1, 'anacol'] # 计算表中的下一个类别就是新类别
        dfres.loc[dfres['trans'] == mincat, 'trans'] = newcat # 替换结果对应表
        dftbl.loc[minindex, 'anacol'] =  newcat # 替换原始表

        # 重新汇总计算表
        dftbl = dftbl.groupby('anacol', as_index = False, 
                              observed = True).agg('sum') # 合并对应的类别
        
        if vartype == 'C': # 对类别变量重新整理出正确顺序，以防万一
            dftbl['badrate'] = dftbl[1] / (dftbl[0] + dftbl[1])
            dftbl.sort_values('badrate', inplace = True)
            dftbl.reset_index(drop = True, inplace = True)
        dftbl.loc[dftbl.index[-1], 'chi2'] = np.NaN # 处理最后一行的卡方值为NA
        if minindex == dftbl.shape[0] - 1: # 合并后已经是最后一行，则调整计算位置
            minindex -= 1
        dftbl.loc[minindex, 'chi2'] = ss.chi2_contingency(dftbl.loc[minindex : minindex + 1,
                                                                    [0, 1]])[0]
        N_levels = dftbl.shape[0] # 更新类别数，避免类别数过少出错

    if needtrans: # 需要将结果转换回原数值对应表
        dfres.columns = ['anacol', 'trans']
        dfres = pd.merge(dfana[[colname, 'anacol']].drop_duplicates(), dfres,
                         how = 'left', on = 'anacol')
        dfres.drop(columns = 'anacol', inplace = True)
    
    dfres.trans = dfres.trans.astype('str') # bin变量强制转换为字符串，保证后续分析不出错
    dfres.columns = [colname, colname + '_bin'] # 标准化输出列名
    
    if len(exclude) > 0: # 将不进行合并的数值写回结果表
        dfres = dfres.append(pd.DataFrame({colname : exclude, colname + '_bin' : exclude}))
        
    if output: # 针对不同的数据类型输出结果
        if vartype == 'N': # 给出每个最终类别的合并范围
            dfresg = dfres.groupby(colname + '_bin')
            dfout = dfresg.min()
            dfout.columns = ['[']
            dfout[']'] = dfresg.max()
            print(dfout)
        elif vartype == 'C': # 给出每个最终类别对应的合并列表
            dftmp = copy.copy(dfres)
            dftmp[colname] = dftmp[colname].astype('str') + ' '
            print(dftmp.groupby(colname + '_bin').sum()) # 这里是sum函数的巧用

    dfres = dfres.set_index(colname).sort_index()
    # 返回指定格式的结果df
    if outtype == 1:
        return dfres
    elif outtype == 2:

def M_CalcWOE(colname, target, output = True):
    # 作者：张文彤；最后修改：2019.5.20
    # 计算分箱后变量的WOE和IV值
    # colname：分箱变量列
    # target：结果变量列，必须为0/1二分类
    # output：是否打印输出描述结果
    # 返回：WOE和IV字典
    
    # 出表前直接做缺失值处理
    dftbl = pd.crosstab(colname.fillna('NA'), target, normalize = 'columns')
    dftbl.columns = ['goodpct', 'badpct']
    dftbl['WOE'] = np.log(dftbl.goodpct / dftbl.badpct) * 100
    IV = sum((dftbl.goodpct - dftbl.badpct) * dftbl.WOE) / 100
    
    if output:
        print("IV : %.3f" % IV)
        dftbl['WOE'].plot.bar()
        plt.gca().set_xticklabels(dftbl.index, rotation = 'vertical', 
                fontproperties = FontProperties(fname = "c:\\windows\\fonts\\STXIHEI.TTF")) 

    return {"WOE": dftbl['WOE'], 'IV':IV}

def M_CompCardValue(dfname, WOEnamelist, paralist, 
                    basescore = 500, baseodds = 1, pdo = 50, nametail = 4):
    ''' 计算各变量各个分箱的信用评分
    dfname
    WOEnamelist：WOE变量列表
    paralist：logistic模型拟合出的回归系数列表
    basescore、baseodds、pdo：计算信用评分时的基本设定
    nametail：需要去除的变量尾部字符数
    ''' 
    
    n = len(WOEnamelist)
    dfres = pd.DataFrame()
    for item in WOEnamelist:
        dftmp = pd.DataFrame({'WOE' : dfname.groupby(item[: - nametail] + '_bin')[item].first()})
        dftmp['varname'] = item
        dfres = dfres.append(dftmp)

    dfres = pd.merge(dfres, pd.DataFrame({'beta' : paralist}), 
                how = 'left', left_on = 'varname', right_index = True)

    factor = pdo/np.log(2)
    offset = basescore - factor * np.log(baseodds)
    dfres['cardscore'] = round(offset / n - factor * (para.Intercept / n + dfres.beta * dfres.WOE))

    dfres['varname'] = dfres['varname'].apply(lambda x : x[: - nametail])
    
    return dfres[['varname', 'cardscore']]

def M_CompKS(score, target, output = True):
    ''' 计算并用图形呈现KS值随分数的变化情况
    score：评分卡分值
    target：是否违约
    '''

    dfana = pd.DataFrame({'target' : target, 'score' : score})
    dfres = pd.crosstab(dfana.score, dfana.target, normalize = 'columns')
    dfres.columns = ['goodpct', 'badpct']
    dfres['goodpct'] = dfres.goodpct.cumsum() * 100
    dfres['badpct'] = dfres.badpct.cumsum() * 100
    dfres['KS'] = dfres.badpct - dfres.goodpct
    if output:
        print(dfres[dfres.KS == dfres.KS.max()]) # 列出ks值最大的一行
        dfres.plot()
    
    return dfres

def M_TimeWindowVar(dfname, idxvar, timegap, catlist, 
                    timewindow = [7, 30, 60, 90, 120, 150, 180], impitem = []):
    ''' 作者：张文彤，最后修改日期：2019.07.10
    基于给定的时间窗口生成分类变量的衍生变量
    dfname 源数据框
    idxvar 用户ID变量名
    timegap 时间差变量名，变量为数值型即可
    catlist 需要转换的分类变量名列表
    timewindow 预定义的时间窗口长度列表
    impitem 需要特别计算是否出现的条目列表
    返回：包含用户ID的结果df
    '''
    
    dfres = dfname[[idxvar]].drop_duplicates()
    
    for tw in timewindow:
        dfana = dfname[dfname[timegap] <= datetime.timedelta(tw).days].copy()
        dfana['cons'] = 1 # 后续操作需要用到常数项
        
        # 计算总操作次数
        cntdict = dfana.groupby([idxvar])['cons'].count().to_dict()
        dfres[catlist[0] + '_' + str(tw) + '_cnt'] = dfres[idxvar].apply(lambda x : 
                                                                     cntdict.get(x, 0))
        for var in catlist:
            # 计算不同取值出现次数
            uniqdict = dfana[[idxvar, 
                              var]].drop_duplicates().groupby([idxvar])[var].count().to_dict()
            dfres[var + '_' + str(tw) + '_uni'] = dfres[idxvar].apply(lambda x : 
                                                                      uniqdict.get(x, 0))
            # 计算各种取值的平均出现次数，这里会出现缺失值
            dfres[var + '_' + str(tw) + '_avg'] = dfres[[catlist[0] + '_' + str(tw) + '_cnt', 
                  var + '_' + str(tw) + '_uni']].apply(lambda x : x[0] * 1.0 / x[1], axis = 1)
            dfres.loc[dfres[var + '_' + str(tw) + '_avg'].isna(), 
                      var + '_' + str(tw) + '_avg'] = 0 # 将缺失值替换为0
            if len(impitem) > 0: # 需要计算特定重要条目的出现次数
                dfres[var + '_' + str(tw) + '_impchg'] = 0 # 特定条目的总出现次数
                for item in impitem: # 标识特定条目是否出现过
                    chgdict = dfana.loc[dfana[var] == item, 
                                [idxvar, 'cons']].groupby([idxvar])['cons'].first().to_dict()  
                    dfres[var + '_' + str(tw) 
                          + item + '_chg'] = dfres[idxvar].apply(lambda x : chgdict.get(x, 0))
                    dfres[var + '_' + str(tw)
                          + '_impchg'] += dfres[var + '_' + str(tw) + item + '_chg']
    return dfres

def M_FillNa(colname, repval = 'mean', naval = []): 
    # 作者：张文彤；最后修改：2019.4.12
    '''有功能扩展的填充缺失值函数
    colname：需要处理的数据列
    repval：替换方式/数值，'mean' 为均数，'random' 为随机替换，其它为指定数值替换方式
    naval, 除NA外需要替换的自定义缺失值数值/字符串，list格式给出
    '''
    TmpLst = copy.copy(colname) # 保护原始数据不被修改

    if len(naval) > 0: # 替换自定义缺失值
        TmpLst = TmpLst.replace(naval, np.nan) 
    
    if repval == 'random':
        for i in range(TmpLst.shape[0]): # 遍历所有数值
            if TmpLst.iloc[i] != TmpLst.iloc[i]: # 如果是缺失值则处理
                while True: # 抽取替代用数值
                    rndrepval = TmpLst.iloc[random.randint(0, TmpLst.shape[0] - 1)]
                    if rndrepval == rndrepval: # 确定已抽出有效值
                        break
                TmpLst.iloc[i] = rndrepval
    elif repval == 'mean':
        TmpLst = TmpLst.fillna(TmpLst.mean())
    else:
        TmpLst = TmpLst.fillna(repval)

    return TmpLst

def M_Inflection(inlst):
    # 返回序列中拐点元素的绝对顺序位置
    if len(inlst) > 3:
        inlst = list(inlst) # 去除可能存在的索引，以免出错
        pointtag = [(inlst[i] < inlst[i + 1] and inlst[i] < inlst[i - 1]) or
                    (inlst[i] > inlst[i + 1] and inlst[i] > inlst[i - 1])
                    for i in range(1, len(inlst) - 1)]
        pointindex = [i + 1 for i in range(len(pointtag)) if pointtag[i]]
    else:
        pointindex = []
        
    return pointindex

def M_MergeBin(bincol, target, exclude = [np.nan], outtype = 1, pctlimit = 0.01, pointlimit = 1):
    ''' 作者：张文彤；最后修改：2019.7.2
    将分箱结果持续合并至单调/单拐点趋势
    bincol 数值型分箱变量列
    exclude 不考虑进行合并的取值列表
    outtype 1 数据替换对应表，2 替换后的完整序列
    pctlimit 单个箱体类别允许的最低比例
    pointlimit 趋势中允许存在的拐点数量，设为0就是只允许单调趋势
    '''
    
    # 结果序列初始化
    dfreplst = pd.DataFrame({"new" : bincol.value_counts(dropna = False)})
    dfreplst.new = dfreplst.index
    
    # 分析集初始化
    dfana = pd.DataFrame({'binvar' : bincol, 'target' : target})
    dfana = dfana[~dfana.binvar.isin(exclude)] # 剔除不需要合并的取值

    while True: # 按pctlimit进行合并
        pctlst = pd.DataFrame({'pct' : dfana.binvar.value_counts(normalize = True, 
                                                                     dropna = False)})
        Inflectpos = [i for i in range(pctlst.shape[0]) 
                          if pctlst.pct.iloc[i] <= pctlimit] # 检索出所有的小比例类别的绝对顺序
        if len(Inflectpos) == 0: # 已不存在小类
            break
        
        dfrep = pd.DataFrame(columns = ['old', 'new'])
        idx = 0
        for i in Inflectpos: # 只考虑和数值相邻的箱体合并
            if i > 0: # 和上一个合并
                dfrep.loc[idx] = [pctlst.index[i], pctlst.index[i - 1]]
                idx += 1
            if i < pctlst.shape[0] - 1: # 和下一个合并
                dfrep.loc[idx] = [pctlst.index[i], pctlst.index[i + 1]]
                idx += 1
        
        for i in dfrep.index: # 计算出每种合并的IV
            dfrep.loc[i, 'IV'] = M_CalcWOE(dfana.binvar.replace(dfrep.loc[i, 'old'],
                     dfrep.loc[i, 'new']), dfana.target, output = False)["IV"]
        
        bestrep = dfrep.sort_values('IV', ascending = False).iloc[0] # 定位最大IV对应的合并方式
        # 按最佳IV进行两个数据框的替换
        dfreplst.new.replace(bestrep['old'], bestrep['new'], inplace = True)
        dfana.binvar = dfana.binvar.replace(bestrep['old'], bestrep['new']) 
    
    while True: # 按pointlimit进行合并
        badratelst = pd.crosstab(dfana.binvar, dfana.target, normalize = 'index')[1]
        Inflectpos = M_Inflection(badratelst) # 定位拐点的绝对位置
        if len(Inflectpos) <= pointlimit: # 单调或者单拐点
            break
        
        dfrep = pd.DataFrame(columns = ['old', 'new'])
        idx = 0
        for i in Inflectpos: # 拐点不可能是序列中最前/最后的位置
            dfrep.loc[idx] = [badratelst.index[i], badratelst.index[i - 1]] # index引用是绝对位置
            dfrep.loc[idx + 1] = [badratelst.index[i], badratelst.index[i + 1]]
            idx += 2
        
        for i in dfrep.index: # 单个样本时这里会报计算错误，但并不影响程序运行
            dfrep.loc[i, 'IV'] = M_CalcWOE(dfana.binvar.replace(dfrep.loc[i, 'old'],
                     dfrep.loc[i, 'new']), dfana.target, output = False)["IV"]
        
        bestrep = dfrep.sort_values('IV', ascending = False).iloc[0] # 定位最大IV对应的合并方式
        # 按最佳IV进行两个数据框的替换
        dfreplst.new.replace(bestrep['old'], bestrep['new'], inplace = True)
        dfana.binvar = dfana.binvar.replace(bestrep['old'], bestrep['new']) 

    if outtype == 1:
        return dfreplst
    else:
        return bincol.replace(dfreplst.new.to_dict())

def M_MergeWOE(dfname, collist, target):
    # 作者：张文彤；最后修改：2019.7.4
    # 批量计算WOE并替换原变量值，该函数比原版本计算速度更快
    # dfname
    # collist：分箱后变量名列表
    # target：因变量名称
    # 返回：所有变量替换为WOE值的df + dfIV
    
    dfres = copy.copy(dfname)
    dfIV = pd.DataFrame()
    for item in collist:
        # print(item) # 用于出错备查
        tmpdict = M_CalcWOE(dfname[item], dfname[target], output = False)
        dfWOE = tmpdict['WOE']
        dfIV.loc[item, 'IV'] = tmpdict["IV"]
        dfWOE = dfWOE.reset_index().replace("NA", np.nan).set_index(item)
        dfres[item].replace(dfWOE.WOE.to_dict(), inplace = True) 
        
    return [dfres, dfIV]

# 自变量和因变量为参数分别指定
# 由于是单变量依次筛选，因此需要先处理掉缺失值，否则结果不准确
import pandas as pd
import numpy as np
import statsmodels.api as sm

def stepwise_selection(X, y, 
                       initial_list = [], 
                       threshold_in = 0.05, 
                       threshold_out = 0.1, 
                       missing = 'drop',
                       verbose = True):
    """ Perform a forward-backward feature selection 
    based on p-value from statsmodels.api.OLS
    Arguments:
        X - pandas.DataFrame with candidate features
        y - list-like with the target
        initial_list - list of features to start with (column names of X)
        threshold_in - include a feature if its p-value < threshold_in
        threshold_out - exclude a feature if its p-value > threshold_out
        verbose - whether to print the sequence of inclusions and exclusions
    Returns: list of selected features 
    Always set threshold_in < threshold_out to avoid infinite looping.
    See https://en.wikipedia.org/wiki/Stepwise_regression for the details
    """
    included = list(initial_list)
    while True:
        changed=False
        # forward step
        excluded = list(set(X.columns)-set(included))
        new_pval = pd.Series(index=excluded)
        for new_column in excluded:
            model = sm.Logit(y, sm.add_constant(
                                pd.DataFrame(X[included+[new_column]])),
                           missing = missing).fit()
            new_pval[new_column] = model.pvalues[new_column]
        best_pval = new_pval.min()
        if best_pval < threshold_in:
            best_feature = new_pval.idxmin()
            included.append(best_feature)
            changed=True
            if verbose:
                print('纳入 {:30} P =  {:.6}'.format(best_feature, best_pval))

        # backward step
        model = sm.Logit(y, sm.add_constant(pd.DataFrame(X[included]))).fit()
        # use all coefs except intercept
        pvalues = model.pvalues.iloc[1:]
        worst_pval = pvalues.max() # null if pvalues is empty
        if worst_pval > threshold_out:
            changed=True
            worst_feature = pvalues.argmax()
            included.remove(worst_feature)
            if verbose:
                print('剔除 {:30} P = {:.6}'.format(worst_feature, worst_pval))
        if not changed:
            break
    return included
